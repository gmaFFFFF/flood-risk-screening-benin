- ! Вычисления весовых коэффициентов проводилось с помощью надстройки PowerQuery Microsoft Excel.
    - ~ Применение алгоритма L1-нормализации вектора приоритетов.
      > Матрица попарного сравнения была развернута в список (денормализована) и над ней был выполнен следующий запрос, соответствующий алгоритму [[Метод анализа иерархий|МАИ]]:

``` powerquery
let
    Источник = ПопарноеСравнение,
    СуммаКоэффициентов = Table.Group(Источник, {"Критерий Правый"}, {{"Сумм", each List.Sum([Коэффициент]), type number}}),
    ДобавитьСумму = Table.NestedJoin(Источник,{"Критерий Правый"},СуммаКоэффициентов,{"Критерий Правый"},"СуммаКоэффициентов",JoinKind.Inner),
    РазвернутьСумму = Table.ExpandTableColumn(ДобавитьСумму, "СуммаКоэффициентов", {"Сумм"}, {"Сумм"}),
    Нормализовать = Table.ReplaceValue(РазвернутьСумму,
                                        each [Коэффициент],
                                        each [Коэффициент]/[Сумм],
                                        Replacer.ReplaceValue,
                                        {"Коэффициент"}),
    #"Удаленные столбцы" = Table.RemoveColumns(Нормализовать,{"Сумм"}),
    #"Сгруппированные строки" = Table.Group(#"Удаленные столбцы", {"Критерий Левый"}, {{"Вес критерия", each List.Average([Коэффициент]), type number}}),
    #"Переименованные столбцы" = Table.RenameColumns(#"Сгруппированные строки",{{"Критерий Левый", "Критерий"}})
in
    #"Переименованные столбцы"
```
